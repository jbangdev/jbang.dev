The jbang way
App.java 
///usr/bin/env jbang "$0" "$@" ; exit $?
//DEPS com.google.code.gson:gson:2.11.0
//JAVA 21+

import com.google.gson.Gson;

void main() {
    var data = Map.of("message", "Hello!");
    System.out.println(new Gson().toJson(data));
}

$ jbang App.java
{"message":"Hello!"}
